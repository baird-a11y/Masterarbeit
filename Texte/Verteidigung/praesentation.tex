\section{Problem Statement}
% ============================================================

\begin{frame}{Problem Statement — Crystal Sedimentation in Magma}
  \begin{columns}[T]
    \column{0.55\textwidth}
      \textbf{Physical Process}
      \begin{itemize}
        \item Crystals (olivine, plagioclase) sediment in viscous magma
        \item Stokes regime: $\mathrm{Re} \ll 1$ (negligible inertia)
        \item Long-range interactions between crystals
        \item Influences thermal and chemical differentiation
      \end{itemize}
      \vspace{0.5em}
      \textbf{Governing Equations}
      \[
        \mathbf{0} = -\nabla p + \mu \nabla^2 \mathbf{u} + \rho\mathbf{g},
        \quad \nabla \cdot \mathbf{u} = 0
      \]

    \column{0.42\textwidth}
      \textbf{Computational Challenge}
      \begin{itemize}
        \item LaMEM solver: ${\approx}\,8$--$9\,\text{s}$ per sample
        \item Scaling: $\mathcal{O}(N^2)$--$\mathcal{O}(N^3)$ with crystal count~$N$
        \item Millions of degrees of freedom for $N > 10$
        \item Systematic parameter studies barely feasible
      \end{itemize}
      \vspace{0.5em}
      \begin{block}{Goal}
        A fast surrogate model that outperforms\\
        the numerical solver by orders of magnitude
      \end{block}
  \end{columns}
\end{frame}

\begin{frame}{Problem Statement — Stream Function as Learning Target}
  \begin{columns}[T]
    \column{0.55\textwidth}
      \textbf{Why the stream function $\psi$?}
      \begin{itemize}
        \item Velocity field derived from $\psi$:
        \[
          u_x = \frac{\partial\psi}{\partial z},\quad
          u_z = -\frac{\partial\psi}{\partial x}
        \]
        \item Incompressibility ($\nabla\!\cdot\!\mathbf{u}=0$) satisfied by construction
        \item Smoother learning target than direct velocity components
        \item Physically consistent predictions by construction
      \end{itemize}

    \column{0.42\textwidth}
      \textbf{Input → Output}
      \begin{itemize}
        \item \textbf{Input:} Geometry mask (crystal positions and radii on a $256\times256$ grid)
        \item \textbf{Output:} Normalized stream function field~$\psi_\text{norm}$
        \item Velocity field derived via finite differences
      \end{itemize}
      \vspace{0.4em}
      \begin{alertblock}{Error Metrics}
        $\bar{\varepsilon}_\psi$: mean relative $L_2$ error on~$\psi$\\
        $\bar{\varepsilon}_v$: mean relative $L_2$ error on velocity
      \end{alertblock}
  \end{columns}
\end{frame}

\section{Scope and Goals}
% ============================================================

\begin{frame}{Scope and Goals}
  \begin{block}{Research Question}
    Can neural surrogate models (FNO and U-Net) predict Stokes flow
    with multiple crystalline inclusions accurately and efficiently?
    Which architecture generalizes better?
  \end{block}

  \vspace{0.5em}
  \begin{columns}[T]
    \column{0.48\textwidth}
      \textbf{In Scope}
      \begin{itemize}
        \item Two architectures: \textcolor{fnocolor}{\textbf{FNO}} and \textcolor{unetcolor}{\textbf{U-Net}}
        \item 4 experiments with increasing complexity
        \item Training \& evaluation on LaMEM-generated data
        \item Hyperparameter search (lr, batch size)
        \item Out-of-distribution (OOD) generalization analysis
      \end{itemize}

    \column{0.48\textwidth}
      \textbf{Out of Scope}
      \begin{itemize}
        \item 3D simulations
        \item Time-dependent flows
        \item Non-stationary boundary conditions
        \item Direct pressure field prediction
        \item Physics-informed loss terms
      \end{itemize}
  \end{columns}

  \vspace{0.5em}
  \begin{columns}[T]
    \column{0.48\textwidth}
      \textbf{FNO:} 100 epochs training
    \column{0.48\textwidth}
      \textbf{U-Net:} 180 epochs training
  \end{columns}
\end{frame}

\section{Architectures: FNO vs.\ U-Net}
% ============================================================

\begin{frame}{Architecture Comparison}
  \begin{columns}[T]
    \column{0.48\textwidth}
      \begin{block}{\textcolor{fnocolor}{Fourier Neural Operator (FNO)}}
        \begin{itemize}
          \item Learns operator $\mathcal{G}: a \mapsto u$ between function spaces
          \item Convolution in the spectral domain (FFT)
          \item Retains only $k_\text{max} = 16$ frequency modes
          \item Global, long-range dependencies
          \item Resolution-independent (by design)
          \item Particularly suited for elliptic PDEs (Stokes)
        \end{itemize}
      \end{block}

    \column{0.48\textwidth}
      \begin{block}{\textcolor{unetcolor}{U-Net}}
        \begin{itemize}
          \item Encoder–decoder with skip connections
          \item Local convolutional kernels (spatial locality)
          \item Multi-scale feature extraction
          \item Strong at geometric structures
          \item Composite loss: MSE + gradient term
          \item High computational efficiency (CPU-native)
        \end{itemize}
      \end{block}
  \end{columns}

  \vspace{0.8em}
  \begin{columns}[T]
    \column{0.48\textwidth}
      \textbf{Advantage:} Global coherence, fewer parameters
    \column{0.48\textwidth}
      \textbf{Advantage:} Spatial detail, faster inference
  \end{columns}

  \vspace{0.5em}
  \begin{center}
    \small\textit{Hypothesis: FNO better for long-range Stokes interactions;
    U-Net stronger at local geometry boundaries}
  \end{center}
\end{frame}

\begin{frame}{Architectures — Key Differences}
  \begin{center}
  \begin{tabular}{lcc}
    \toprule
    \textbf{Property} & \textcolor{fnocolor}{\textbf{FNO}} & \textcolor{unetcolor}{\textbf{U-Net}} \\
    \midrule
    Learning paradigm    & Operator (function space) & Image-to-image \\
    Receptive field      & Global (spectral)          & Local (convolution) \\
    Frequency processing & Selective ($k \leq k_\text{max}$) & Implicit (pooling) \\
    High-frequency       & Limited ($k_\text{max}=16$) & Better (skip conn.) \\
    Boundary conditions  & Softly enforced            & Directly in geometry \\
    Inference time       & ${\approx}687\,\text{ms}$ (incl. transfer) & ${\approx}5{.}5\,\text{ms}$ \\
    Speedup vs.\ LaMEM  & ${\approx}12\times$         & ${>}1500\times$ \\
    \bottomrule
  \end{tabular}
  \end{center}

  \vspace{0.5em}
  \begin{alertblock}{Spectral Limitation of FNO}
    The FNO discards Fourier modes with $k > k_\text{max} = 16$. For flows
    with sharp boundary layers (large crystals) this leads to systematic
    approximation errors.
  \end{alertblock}
\end{frame}

\section{Experiments}
% ============================================================

\begin{frame}{Experimental Design — Overview}
  \begin{columns}[T]
    \column{0.52\textwidth}
      \textbf{Dataset (LaMEM-generated)}
      \begin{itemize}
        \item Domain: $[-1,1]^2$, grid $256\times256$
        \item Fixed: crystal radius $r = 0.05$
        \item 1\,000 train / 100 validation / 10 evaluation (per stratum)
        \item Normalized stream function as target
      \end{itemize}

      \vspace{0.8em}
      \textbf{Hyperparameter Search (Exp.~1)}
      \begin{itemize}
        \item LR: $10^{-3},\; 5\!\times\!10^{-3},\; 10^{-4},\; 5\!\times\!10^{-4}$
        \item Batch size: 8 and 16
        \item $\Rightarrow$ 8 configurations per architecture
      \end{itemize}

    \column{0.45\textwidth}
      \begin{enumerate}
        \item[\textbf{Exp.~1}] \textbf{Single-Crystal Baseline}\\
          $n = 1$, hyperparameter search
        \vspace{0.4em}
        \item[\textbf{Exp.~2}] \textbf{Multi-Crystal Generalization}\\
          Training: $n \in \{1,\ldots,10\}$\\
          OOD: $n \in \{11, 12, 15, 20, 25\}$
        \vspace{0.4em}
        \item[\textbf{Exp.~3}] \textbf{Stress Test}\\
          Training: $n \leq 25$\\
          OOD: $n \in \{35, 50, 60, 80, 84\}$
        \vspace{0.4em}
        \item[\textbf{Exp.~4}] \textbf{Crystal Size}\\
          OOD radii: $r \in \{0.005, 0.5\}$\\
          Variable-radius FNO: $r \in \{0.02, 0.05, 0.08\}$
      \end{enumerate}
  \end{columns}
\end{frame}

\begin{frame}{Results — FNO Dominance and Its Limits}
  \begin{columns}[T]
    \column{0.48\textwidth}
      \textbf{\textcolor{fnocolor}{FNO Strengths}}
      \begin{itemize}
        \item Clearly better in-distribution (factor $3$–$4\times$)
        \item Better OOD generalization to more crystals
        \item No overfitting over 100 epochs
        \item Monotone error convergence
        \item Spatially uniform error distribution
      \end{itemize}

      \vspace{0.5em}
      \textbf{\textcolor{unetcolor}{U-Net Strengths}}
      \begin{itemize}
        \item $1500\times$ faster than LaMEM
        \item More robust on large crystals (Exp.~4)
        \item OOD error grows slower at extreme $n$
        \item Easier to deploy on CPU
      \end{itemize}

    \column{0.48\textwidth}
      \textbf{Critical Limitations}
      \begin{itemize}
        \item \textbf{FNO:} Spectral bias at $k > k_\text{max}$\\
          → catastrophic failure on large crystals ($71\%$)
        \item \textbf{U-Net:} Generally higher base error\\
          → still behind despite 180 epochs
        \item Both: $n>50$ only limitedly reliable
        \item Very small evaluation sets ($n=10$ samples)
      \end{itemize}

      \vspace{0.5em}
      \begin{block}{Conclusion}
        FNO is the stronger architecture for this task,
        as long as no extreme geometry scales occur.
      \end{block}
  \end{columns}
\end{frame}

\begin{frame}{Backup — FNO Architecture Details}
  \begin{itemize}
    \item Fourier layer: Input $\to$ FFT $\to$ weights on $k \leq k_\text{max}$ modes $\to$ IFFT $+ W\cdot v$
    \item $k_\text{max} = 16$, 4 Fourier layers, $d_v = 64$ channels
    \item Activation function: GELU
    \item Adam optimizer, 100 epochs, MSE loss
    \item Resolution: $256 \times 256$ (resolution-independent by design)
  \end{itemize}
\end{frame}

\begin{frame}{Backup — U-Net Architecture Details}
  \begin{itemize}
    \item Encoder: 4 levels, each Conv-BN-ReLU, MaxPool
    \item Decoder: transposed conv + skip connections
    \item Composite loss: MSE + $\lambda \cdot \|\nabla(\hat\psi - \psi)\|^2$
    \item Adam optimizer, 180 epochs
    \item Best-checkpoint selection via validation $\bar{\varepsilon}_\psi$
  \end{itemize}
\end{frame}

\begin{frame}{Backup — Metrics in Detail}
  \begin{itemize}
    \item $\bar{\varepsilon}_\psi = \frac{1}{N}\sum_{i=1}^{N} \frac{\|\hat\psi_i - \psi_i\|_2}{\|\psi_i\|_2}$: mean relative $L_2$ error on $\psi$
    \item $\bar{\varepsilon}_v$: analogous for derived velocity field
    \item $\bar{e}_\text{max}$: mean maximum absolute error on $\psi$
    \item $\overline{\text{MSE}}_\psi$: mean squared error on $\psi$
    \item Divergence RMS: physical sanity check ($\sim10^{-26}$ for FNO)
  \end{itemize}
